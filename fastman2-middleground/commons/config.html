配置

配置

框架内置了很多基础的插件，通过简单配置即可实现运用。

会话配置

module.exports = appInfo => ({
  // 会话配置
  session: {
    // 存储该Session的Cookie键值对名称
    key: 'FS2_SESS',
    // 设置这个键值对在浏览器的最长保存时间，默认20分钟
    maxAge: 20 * 60 * 1000,
    // 设置键值对是否可以被js访问，默认为true，不允许被js访问
    httpOnly: true,
    // 设置是否对 Cookie 进行加密，如果设置为 true，则在发送 Cookie 前会对这个键值对的值进行加密，客户端无法读取到 Cookie 的明文值。默认为 false
    encrypt: false,
    // 延长有效期，一旦开启，当有效期仅剩下最大有效期一般时重置Session的有效期
    renew: false,
  },
});

redis配置

框架默认的Session存储机制是基于Redis来进行扩展存储的。 

module.exports = appInfo => ({
  // redis配置
  redis: {
    client: {
      // Redis端口
      port: 6379,
      // Redis所属服务器IP
      host: '127.0.0.1',
      // Redis密码
      password: '',
      // Redis数据库索引，一共有15个，从0开始计数
      // 不同的数据类型可通过不同的redis数据源来存储不同的db，从而实现良好的数据治理
      db: 0,
    },
  },
});

日志配置

// config/config.${env}.js
module.exports = appInfo => ({
  // 默认日志配置
  logger: {
    // 日志文件存放目录，在本地开发环境 (env: local) 和单元测试环境 (env: unittest)，为了避免冲突以及集中管理，日志会打印在项目目录下的 logs 目录，而在其它环境下建议存放在服务器的指定目录下
    dir: `${appInfo.root}/logs/${appInfo.name}`,
    // 默认编码，或可以指定为gbk
    encoding: 'utf-8',
    // 允许输出的最小级别日志级别，可选值：NONE/DEBUG/INFO/WARN/ERROR
    level: 'DEBUG', // 即DEBUG及以上（INFO/WARN/ERROR）的日志都会打印
    // 允许输出的最小级别中断日志级别，可选值：NONE/DEBUG/INFO/WARN/ERROR
    consoleLevel: 'DEBUG',
    // 应用相关日志文件名
    appLogName: `${appInfo.name}-web.log`,
    // 框架内核、插件日志文件名
    coreLogName: 'fastman2-moddleground-web.log',
    // agent进程日志文件名
    agentLogName: 'fastman2-moddleground-agent.log',
    // 任意logger的.error()调用输出的日志文件名
    errorLogName: 'common-error.log',
    // // 为了避免一些插件的调试日志在生产环境打印导致性能问题，生产环境默认禁止答应DEBUG级别日志，如需打印则开启该配置项
    // allowDebugAtProd: true,
  },

  // 自定义日志配置
  customLogger: {
    scheduleLogger: {
      // consoleLevel: 'NONE',
      file: path.join(appInfo.root, 'logs', appInfo.name, 'fastman2-moddleground-schedule.log'),
    },
    // 可通过app.getLogger('xxLogger') / ctx.getLogger('xxLogger') 获取，最终的打印结果和 coreLogger 类似
    XXXLogger: {
      file: path.join(appInfo.root, 'logs/xx.log'),
    },
  },

  // 日志切割配置
  logrotator: {
    // 按照大小来切割日志文件，添加到filesRotateBySize的日志文件将不再按天进行切割
    filesRotateBySize: [
      path.join(appInfo.root, 'logs', appInfo.name, 'fastman2-moddleground-web.log'),
    ],
    // 超过该大小设置则开始切割
    maxFileSize: 2 * 1024 * 1024 * 1024,
    // 按照小时来切割日志文件，添加到 filesRotateByHour 的日志文件不再被按天进行切割
    filesRotateByHour: [
      path.join(appInfo.root, 'logs', appInfo.name, 'common-error.log'),
    ],
  },
});

用户认证配置

module.exports = appInfo => ({
  jwt: {
    // 是否启动，默认关闭
    enable: false,
    // 密钥定义，主要用于签名
    secret: 'xxx',
    // Token失效时间，可使用具体的单位来约束时间（如120ms、2 days、10h、7d）,如果直接填写数值则默认单位为ms，不设置默认为30分钟
    expiresIn: '0.5h',
  },
});

数据库配置

框架推荐使用mysql作为数据存储，如果想使用其它数据库，框架内部提供了非常多的数据源插件。 

module.exports = appInfo => ({
  mysql: {
    client: {
      // 主机ip
      host: 'localhost',
      // 端口号
      port: '3306',
      // 用户名
      user: 'xxx',
      // 密码
      password: 'xxx',
      // 数据库名
      database: 'fastman',
    },
    // 是否加载到 app 上，默认开启
    app: true,
    // 是否加载到 agent 上，默认关闭
    agent: false,
  },
});
