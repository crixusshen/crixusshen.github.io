面向切面

AOP

面向切面编程，可将代码逻辑植入到对应的函数位置进行执行，这些代码逻辑的执行都是可以被控制的。

在了解AOP之前首先需要了解几个基本概念：

  • Aspect（切面）：通常是一个类，里面可定义切入点和通知
  • JointPoint（连接点）：程序执行过程中明确的点，一般是方法的调用
  • Advice（通知）：AOP在特定的切入点上执行的增强处理，有before，after，afterReturning，afterThrowing，around
  • Pointcut（切入点）：就是带有通知的连接点，在程序中主要体现为书写切入点表达式

Around

环绕通知,在目标方法完成前后做增强处理,环绕通知是最重要的通知类型,像事务,日志等都是环绕通知。 

const aop = require('fastman2-middleground-framework').Aop;

aop.around('target instance', 'method name', function(joinpoint){
  // 可通过joinpoint获取目标实例
  const targetInstance = joinpoint.target;
  // 同时可通过实例获取方法调用的参数
  const args = targetInstance.args;
  // 如果函数非箭头函数，也可通过this获取到对应的实例
  assert.equal(this, targetInstance); // ture

  // TODO 这里可以实现目标方法完成前的增强处理 ...

  // 调用目标方法
  const returnVal = joinpoint.proceed();

  // TODO 这里可以实现目标方法完成后的增强处理 ...

});

同样还能支持嵌套环绕，最外层的环绕会先被执行，直到遇到了joinpoint.proceed才会执行内层的环绕，当内层执行完毕后才会执行外层剩余的逻辑体： 

it('should be invoked from most recently added to least recently added', () => {
  const inner = buster.sinon.spy();
  const target = new Fixture();
  // Inner
  aop.around(target, 'method', joinpoint => {
    inner();
    joinpoint.proceed();
  });
  // Outer
  aop.around(target, 'method', joinpoint => {
    refute.calledOnce(inner); // inner()还未被执行
    joinpoint.proceed();
    buster.assert.calledOnce(inner); // inner()被执行了一次
  });
  target.method(arg);
});

有时我们想在增强处理后对原执行函数的参数做一些修改，通过下面的方式就可以实现： 

it('should be able to modify arguments', () => {
  const target = new Fixture();
  const spy = target.method = buster.sinon.spy();

  // Starting value
  assert.equal(0, target.val);

  aop.around(target, 'method', joinpoint => {
    // arg should be the return value from the orig method
    assert.equal(1, joinpoint.args.length);
    assert.equal(arg, joinpoint.args[0]);

    // Modify the original args and pass them through to
    // the original func
    return joinpoint.proceed(1, 2, 3);

    // proceed(...args) and proceedApply([args]) are equivalent
    // return joinpoint.proceedApply([ 1, 2, 3 ]);
  });

  target.method(arg);

  // Make sure the return value is preserved, also based on the modified args
  buster.assert.calledOnceWith(spy, 1, 2, 3);
});

Before

前置通知，在目标方法被调用之前做增强处理。 

const aop = require('fastman2-middleground-framework').Aop;
it('should invoke advice before advised method', () => {
  const target = new Fixture();

  aop.before(target, 'method', function beforeAdvice(a) {
    // beforeAdvice必须是一个包裹函数，而不能是箭头函数，否则拿不到this，此时的this是一个joinpoint实例
    assert.equal(target, this);

    // beforeAdvice的参数及调用函数的args
    assert.equal(arg, a);

    // 在调用函数还未被调用之前，内部变量应该没有发生变化
    assert.equal(0, this.val);
  });

  const ret = target.method(arg);
  assert.equal(1, target.val);
  assert.equal(ret, target.val);
});

同样前置通知也支持嵌套声明，当声明多个前置通知后，最近一次被声明的前置通知总是会优先被调用，这一点同Around通知。 

it('should invoke most recently added advices first', () => {
  const target = new Fixture();
  let beforeCount = 0;

  // 添加3个前置通知来测试他们的调用顺序、参数和返回值
  aop.before(target, 'method', function before1(a) {
    assert.equal(arg, a);
    assert.equal(0, this.val);
    assert.equal(2, beforeCount);
  });

  aop.before(target, 'method', function before2(a) {
    assert.equal(arg, a);
    assert.equal(0, this.val);
    assert.equal(1, beforeCount);
    beforeCount++;
  });

  aop.before(target, 'method', function before3(a) {
    assert.equal(arg, a);
    assert.equal(0, this.val);
    assert.equal(0, beforeCount);
    beforeCount++;
  });

  const ret = target.method(arg);
  assert.equal(1, target.val);
  assert.equal(ret, target.val);
});

After

后置通知，在目标方法完成之后做增强，无论目标方法是否成功完成。 

it('should invoke advice after advised method', () => {
  const target = new Fixture();

  assert.equal(0, target.val);

  aop.after(target, 'method', function afterAdvice(a) {
    assert.equal(target, this);
    // 与beforeAdvice不同的是，这里的a不再是函数调用参数，而是被增强的函数调用后的返回值
    assert.equal(this.val, a);
    assert.equal(1, this.val);
  });

  const ret = target.method(arg);

  assert.equal(1, target.val);
  assert.equal(ret, target.val);
});

当被增强的函数出现了异常情况，则afterAdvice(arg)中的arg将会是一个Error对象： 

it('should invoke advice after thrown exception', () => {
  const target = new Fixture(true);

  assert.equal(0, target.val);

  aop.after(target, 'method', function afterAdvice(a) {
    assert.equal(target, this);

    // 当被增强的函数发生了异常，则arg是一个Error对象
    assert(a instanceof Error);

    assert.equal(1, this.val);
  });

  buster.assert.exception(function() {
    target.method(arg);
  });

  assert.equal(1, target.val);
});

AfterReturning

在目标方法正常完成后做增强。 

it('should invoke advice after advised method returns', () => {
  const target = new Fixture();

  assert.equal(0, target.val);

  aop.afterReturning(target, 'method', function(arg) {
    assert.equal(target, this);
    // arg应该是被增强函数的返回值
    assert.equal(this.val, arg);
    assert.equal(1, this.val);
  });

  const ret = target.method(arg);
  assert.equal(1, target.val);
  assert.equal(ret, target.val);
});

不同于其它通知，after*通知都是从内层开始向外层开始调用的，因此内层的通知会首先被调用，外层最后被调用： 

it('should invoke most recently added advices last', () => {
  const target = new Fixture();
  let count = 0;

  assert.equal(0, target.val);

  aop.afterReturning(target, 'method', function afterReturning1(a) {
    assert.equal(target, this);
    assert.equal(this.val, a);
    assert.equal(1, this.val);
    assert.equal(0, count);
    count++;
  });

  aop.afterReturning(target, 'method', function afterReturning2(a) {
    assert.equal(target, this);
    assert.equal(this.val, a);
    assert.equal(1, this.val);
    assert.equal(1, count);
    count++;
  });

  aop.afterReturning(target, 'method', function afterReturning3(a) {
    assert.equal(target, this);
    assert.equal(this.val, a);
    assert.equal(1, this.val);
    assert.equal(2, count);
  });

  const ret = target.method(arg);
  assert.equal(1, target.val);
  assert.equal(ret, target.val);
});

当被增强函数发生异常，afterReturing通知是不会被调用的，因为该通知只会发生在成功完成后。 

it('should not be invoked when original method throws', () => {
  const target = new Fixture();
  target.method = buster.sinon.stub().throws(new Error());

  const spy = buster.sinon.spy();

  aop.afterReturning(target, 'method', spy);

  buster.assert.exception(function() {
    target.method(arg);
  });

  refute.called(spy);
});

AfterThrowing

主要用来处理程序中未处理的异常。 

it('should invoke advice after advised method throws', () => {
  const target = new Fixture(true);

  assert.equal(0, target.val);

  aop.afterThrowing(target, 'method', function afterThrowing(a) {
    assert.equal(target, this);
    assert(a instanceof Error);
    assert.equal(1, this.val);
  });

  buster.assert.exception(function() {
    target.method(arg);
  });

  assert.equal(1, target.val);
});

当被增强函数出现异常时，如果afterThrowing通知被注册则将会被调用。 

it('should not be invoked when original returns without throwing', () => {
  const target = new Fixture();
  target.method = buster.sinon.stub().returns(123);

  const spy = buster.sinon.spy();

  aop.afterThrowing(target, 'method', spy);

  let ret;
  refute.exception(function() {
    ret = target.method(arg);
  });

  refute.called(spy);
  assert.equal(ret, 123);
});

Pointcut

有时候针对同一个函数多处需要增加的地方我们不需要重复的去定义，可通过下面的表达式简单来定义： 

it('array should specify exact list of method names', () => {
  let method1,
    method2,
    method3;

  const target = {
    method1: (method1 = buster.sinon.spy()),
    method2: (method2 = buster.sinon.spy()),
    method3: (method3 = buster.sinon.spy()),
  ;

  const before = buster.sinon.spy();

  // [ 'method1', 'method3' ]也可以使用正则表达式/method[13]/来代替。
  // 也可以使用一个具有返回值的函数来代替，例如：
  // function() { return [ 'method1', 'method3' ]; }
  aop(target, [ 'method1', 'method3' ], {
    before,
  });

  target.method1();
  buster.assert.calledOnce(before);
  buster.assert.calledOnce(method1);
  refute.called(method2);
  refute.called(method3);

  target.method2();
  refute.calledTwice(before);
  buster.assert.calledOnce(method2);
  refute.called(method3);

  target.method3();
  buster.assert.calledTwice(before);
  buster.assert.calledOnce(method3);
});
